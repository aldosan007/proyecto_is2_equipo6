  CREATE TABLE "DEMO_SDLC" 
   (	"SDLCNO" NUMBER(8,0), 
	"SDLCNAME" VARCHAR2(15), 
	"MAX_WIP" NUMBER(8,0), 
	"CTROL_VENCIMIENTO" VARCHAR2(250),

   ) ;

  ALTER TABLE "DEMO_SDLC" ADD CONSTRAINT "DEMO_SDLC_FK_TABLERO" FOREIGN KEY ("COD_TABLERO")

  CREATE OR REPLACE EDITIONABLE TRIGGER "TRG_DEMO_SDLC_ID" 
BEFORE INSERT ON DEMO_SDLC
FOR EACH ROW
BEGIN
   -- Si no se especifica un ID en el INSERT, toma el siguiente valor de la secuencia
   IF :NEW.SDLCNO IS NULL THEN
      :NEW.SDLCNO := PROGRESO_TAREA.NEXTVAL;
   END IF;
END;
/
ALTER TRIGGER "TRG_DEMO_SDLC_ID" ENABLE;
